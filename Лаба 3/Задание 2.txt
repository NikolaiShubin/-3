def find_common_participants(first, second, a =","):
    participants_first_group = first.split(a)

    participants_second_group = second.split(a)

    common = []

    for x in participants_first_group:

        if x in participants_second_group:

            common.append(x)

    return sorted(common)

participants_first_group = "Иванов|Петров|Сидоров"
participants_second_group = "Петров|Сидоров|Смирнов"

y = find_common_participants(participants_first_group, participants_second_group)
print("Общие участники:", y)
